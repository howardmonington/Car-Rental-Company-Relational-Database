Query Commands
 
1.	SELECT LASTNAME, DROPOFFLOCATION
		FROM PERSON, RENTALTRANSACTION
		WHERE PERSON.LICENSEID = RENTALTRANSACTION.LICENSEID
			AND DROPOFFLOCATION IS NULL
			ORDER BY LASTNAME ASC;

Results: 
LASTNAME: Monington 
DROPOFFLOCATION: null

2.	SELECT MAKE, COLOR, DROPOFFLOCATION
		FROM CAR, RENTALTRANSACTION 
		WHERE CAR.VIN = RENTALTRANSACTION.VIN 
			AND DROPOFFLOCATION IS NULL; 

Results:
MAKE: Tesla
COLOR: Blue
DROPOFFLOCATION: null 

3.	SELECT RENTALTRANSACTION.TRANSID, PRICE
		FROM RENTALTRANSACTION, CARCLASSIFICATION, CAR
		WHERE RENTALTRANSACTION.VIN = CAR.VIN
			AND CAR.CCLASS = CARCLASSIFICATION.CCLASS 
			AND DROPOFFLOCATION IS NOT NULL;

Results:

45467776	75
52264488	100
88597622	200

5.	SELECT FIRSTNAME, LASTNAME, FIRSTPURCHASEDATE, POSITIONTITLE
 		FROM PERSON

6.	SELECT MIN(PRICE)
    		FROM CARCLASSIFICATION
    		WHERE PRICE IN
        		(SELECT PRICE	
            			FROM RENTALTRANSACTION, CARCLASSIFICATION, CAR
            			WHERE RENTALTRANSACTION.VIN = CAR.VIN
                			AND CAR.CCLASS = CARCLASSIFICATION.CCLASS 
                			AND DROPOFFLOCATION IS NOT NULL);

Result:
MIN(PRICE): 75

7.	SELECT MAKE, MODEL, CAR.VIN
		FROM CAR
		WHERE CAR.VIN NOT IN 
			(SELECT CAR.VIN
				FROM CAR, RENTALTRANSACTION
				WHERE CAR.VIN = RENTALTRANSACTION.VIN);
