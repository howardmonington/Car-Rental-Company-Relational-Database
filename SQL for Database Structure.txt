SQL for Database Structure

Create Table PROMOTION
(DISCOUNTCODE VARCHAR2 (2) PRIMARY KEY,
DISCOUNTTYPE VARCHAR2 (12),
DISCOUNTDESC VARCHAR2 (30));


Create Table STORELOCATION
(STOREID VARCHAR2 (9) PRIMARY KEY,
STREET VARCHAR2 (30),
CITY VARCHAR2 (30),
USSTATE CHAR (2),
POSTALCODE CHAR (9));

Create Table CARCLASSIFICATION
(CCLASS VARCHAR2 (10) PRIMARY KEY,
TOLLPASS VARCHAR2 (3),
PRICE NUMBER (8),
DISCOUNTCODE VARCHAR2 (2), 
FOREIGN KEY (DISCOUNTCODE) REFERENCES PROMOTION(DISCOUNTCODE));

Create Table CAR
(VIN VARCHAR (17) PRIMARY KEY,
YEAR CHAR (4),
COLOR VARCHAR2 (30),
MAKE VARCHAR2 (30),
LICENSEPLATE VARCHAR2 (8),
MODEL VARCHAR2 (30),
STOREID VARCHAR2 (5),
CCLASS VARCHAR2 (10),
FOREIGN KEY (CCLASS) REFERENCES CARCLASSIFICATION(CCLASS),
FOREIGN KEY (STOREID) REFERENCES STORELOCATION(STOREID));


Create Table PERSON
(LICENSEID VARCHAR2 (9) PRIMARY KEY,
FIRSTNAME VARCHAR2 (30) NOT NULL,
LASTNAME VARCHAR2 (30) NOT NULL,
STREET VARCHAR2 (30) NOT NULL,
CITY VARCHAR2 (30) NOT NULL,
USSTATE CHAR (2) NOT NULL,
POSTALCODE CHAR (9) NOT NULL,
PHONE1 CHAR (10) NOT NULL,
PHONE2 CHAR (10),
MEMBERSHIPLVL VARCHAR2 (15),
FIRSTPURCHASEDATE DATE,
CUSTOMERTYPE VARCHAR2 (2),
POSITIONTYPE VARCHAR2 (20),
POSITIONTITLE VARCHAR2 (20),
BENEFITS VARCHAR2 (12),
STARTDATE DATE,
STOREID VARCHAR2 (5),
FOREIGN KEY (STOREID) REFERENCES STORELOCATION(STOREID)); 

Create Table RENTALTRANSACTION
(TRANSID NUMBER (8) PRIMARY KEY,
PICKUPLOCATION VARCHAR2 (30),
DROPOFFLOCATION VARCHAR2 (30),
BEGINNINGODOMETER NUMBER (8),
ENDODOMETER NUMBER (8),
ENDTANK VARCHAR2 (17),
RENTALDATE DATE,
RENTALPOLICY VARCHAR2 (2),
LICENSEID VARCHAR2 (9),
VIN VARCHAR2 (17),
FOREIGN KEY (LICENSEID) REFERENCES PERSON(LICENSEID),
FOREIGN KEY (VIN) REFERENCES CAR(VIN));
